CREATE TABLE Department
(
  department.Id CHAR NOT NULL,
  departmentName VARCHAR NOT NULL,
  PRIMARY KEY (department.Id)
);

CREATE TABLE Medicine
(
  m.Id CHAR NOT NULL,
  name VARCHAR NOT NULL,
  price NUMERIC NOT NULL,
  PRIMARY KEY (m.Id)
);

CREATE TABLE Doctor
(
  d.Id CHAR NOT NULL,
  firstName VARCHAR NOT NULL,
  lastName VARCHAR NOT NULL,
  birthDay DATE NOT NULL,
  gender CHAR NOT NULL,
  department.id CHAR NOT NULL,
  salary NUMERIC NOT NULL,
  bonus NUMERIC,
  department.Id CHAR NOT NULL,
  PRIMARY KEY (d.Id),
  FOREIGN KEY (department.Id) REFERENCES Department(department.Id)
);

CREATE TABLE Nurse
(
  n.Id CHAR(4) NOT NULL,
  firstName VARCHAR(20) NOT NULL,
  lastName VARCHAR(20) NOT NULL,
  birthDAy DATE NOT NULL,
  gender CHAR(1) NOT NULL,
  salary NUMERIC(9,2) NOT NULL,
  bonus NUMERIC(9,2),
  department.Id CHAR NOT NULL,
  PRIMARY KEY (n.Id),
  FOREIGN KEY (department.Id) REFERENCES Department(department.Id)
);

CREATE TABLE Room
(
  r.Id CHAR NOT NULL,
  n.Id CHAR(4) NOT NULL,
  PRIMARY KEY (r.Id),
  FOREIGN KEY (n.Id) REFERENCES Nurse(n.Id)
);

CREATE TABLE mentors
(
  n.Id CHAR(4) NOT NULL,
  d.Id CHAR NOT NULL,
  PRIMARY KEY (n.Id, d.Id),
  FOREIGN KEY (n.Id) REFERENCES Nurse(n.Id),
  FOREIGN KEY (d.Id) REFERENCES Doctor(d.Id)
);

CREATE TABLE train
(
  n.Id_1 CHAR(4) NOT NULL,
  trainn.Id_2 CHAR(4) NOT NULL,
  PRIMARY KEY (n.Id_1, trainn.Id_2),
  FOREIGN KEY (n.Id_1) REFERENCES Nurse(n.Id),
  FOREIGN KEY (trainn.Id_2) REFERENCES Nurse(n.Id)
);

CREATE TABLE Doctor_specialize
(
  specialize VARCHAR NOT NULL,
  d.Id CHAR NOT NULL,
  PRIMARY KEY (d.Id),
  FOREIGN KEY (d.Id) REFERENCES Doctor(d.Id)
);

CREATE TABLE fPatient
(
  p.Id CHAR NOT NULL,
  firstName VARCHAR NOT NULL,
  lastName VARCHAR NOT NULL,
  gender CHAR NOT NULL,
  birthDay DATE NOT NULL,
  SSN CHAR NOT NULL,
  d.Id CHAR NOT NULL,
  n.Id CHAR(4) NOT NULL,
  r.Id CHAR NOT NULL,
  PRIMARY KEY (p.Id),
  FOREIGN KEY (d.Id) REFERENCES Doctor(d.Id),
  FOREIGN KEY (n.Id) REFERENCES Nurse(n.Id),
  FOREIGN KEY (r.Id) REFERENCES Room(r.Id),
  UNIQUE (SSN)
);

CREATE TABLE Record
(
  record.Id CHAR NOT NULL,
  description VARCHAR NOT NULL,
  p.Id CHAR NOT NULL,
  d.Id CHAR NOT NULL,
  PRIMARY KEY (record.Id, p.Id),
  FOREIGN KEY (p.Id) REFERENCES fPatient(p.Id),
  FOREIGN KEY (d.Id) REFERENCES Doctor(d.Id)
);

CREATE TABLE Bill
(
  b.Id CHAR NOT NULL,
  insuaranceDeduct NUMERIC NOT NULL,
  date DATE NOT NULL,
  p.Id CHAR NOT NULL,
  PRIMARY KEY (b.Id, p.Id),
  FOREIGN KEY (p.Id) REFERENCES fPatient(p.Id)
);

CREATE TABLE listed_in
(
  quantity INT NOT NULL,
  b.Id CHAR NOT NULL,
  p.Id CHAR NOT NULL,
  m.Id CHAR NOT NULL,
  PRIMARY KEY (b.Id, p.Id, m.Id),
  FOREIGN KEY (b.Id, p.Id) REFERENCES Bill(b.Id, p.Id),
  FOREIGN KEY (m.Id) REFERENCES Medicine(m.Id)
);

CREATE TABLE Patient_address
(
  address VARCHAR NOT NULL,
  p.Id CHAR NOT NULL,
  PRIMARY KEY (address, p.Id),
  FOREIGN KEY (p.Id) REFERENCES fPatient(p.Id)
);

CREATE TABLE Patient_phoneNumber
(
  phoneNumber CHAR NOT NULL,
  p.Id CHAR NOT NULL,
  PRIMARY KEY (phoneNumber, p.Id),
  FOREIGN KEY (p.Id) REFERENCES fPatient(p.Id)
);
